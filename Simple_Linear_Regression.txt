# Step 1: Import required libraries
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_absolute_error, mean_absolute_percentage_error, mean_squared_error
import matplotlib.pyplot as plt

# Step 2: Load dataset
data = pd.read_csv('https://raw.githubusercontent.com/selva86/datasets/master/Salary_Data.csv')

# Step 3: Define target (y) and feature (X)
X = data[['YearsExperience']]
y = data['Salary']

# Step 4: Split the data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(X, y, train_size=0.7, random_state=2529)

# Step 5: Create and train the Linear Regression model
model = LinearRegression()
model.fit(X_train, y_train)

# Step 6: Display model parameters
print("ğŸ“ˆ Simple Linear Regression Model (Salary vs Experience)")
print(f"Intercept (bâ‚€): {model.intercept_:.2f}")
print(f"Coefficient (bâ‚): {model.coef_[0]:.2f}")

# Step 7: Predict using the test data
y_pred = model.predict(X_test)

# Step 8: Evaluate model performance
print("\nğŸ” Model Evaluation:")
print(f"Mean Absolute Error (MAE): {mean_absolute_error(y_test, y_pred):.2f}")
print(f"Mean Absolute Percentage Error (MAPE): {mean_absolute_percentage_error(y_test, y_pred)*100:.2f}%")
print(f"Mean Squared Error (MSE): {mean_squared_error(y_test, y_pred):.2f}")

# Step 9: Visualization
plt.scatter(X, y, color='blue', label='Actual Data')
plt.plot(X, model.predict(X), color='red', label='Regression Line')
plt.xlabel('Years of Experience')
plt.ylabel('Salary')
plt.title('Salary vs Experience (Linear Regression)')
plt.legend()
plt.show()

# Step 10: User input for prediction
print("\nğŸ’¬ Predict Salary Based on Your Experience")

try:
    user_exp = float(input("Enter your experience in years: "))
    predicted_salary = model.predict([[user_exp]])
    print(f"ğŸ’¼ Predicted Salary for {user_exp:.1f} years of experience: â‚¹{predicted_salary[0]:,.2f}")
except ValueError:
